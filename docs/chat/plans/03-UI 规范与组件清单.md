### 9.2 API 接口设计（Repository Protocol）

```swift
protocol ThoughtRepositoryProtocol {
    // MARK: - Fetch
    func fetchAll(limit: Int, offset: Int, sortBy: ThoughtSortOption) async throws -> [Thought]
    func fetchById(_ id: UUID) async throws -> Thought?
    func fetchByTag(_ tagName: String) async throws -> [Thought]
    func fetchByMood(_ mood: Mood) async throws -> [Thought]
    func search(query: String, filters: ThoughtFilters) async throws -> [Thought]

    // MARK: - CRUD
    func create(content: String, mood: Mood?, tags: [String], image: Data?) async throws -> Thought
    func update(_ id: UUID, content: String?, mood: Mood?, tags: [String]?) async throws -> Thought
    func delete(_ id: UUID) async throws

    // MARK: - References
    func addReference(sourceId: UUID, targetId: UUID) async throws
    func removeReference(sourceId: UUID, targetId: UUID) async throws
    func getReferences(for id: UUID) async throws -> [Thought]
    func getReferencedBy(id: UUID) async throws -> [Thought]

    // MARK: - Tags
    func getAllTags() async throws -> [ThoughtTag]
    func mergeTags(oldName: String, newName: String) async throws
}
```
